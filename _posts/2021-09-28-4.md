# call by value 예

```c
#include <stdio.h>
int sum(int x, int y);
int main(void)
{
	int a = 10, b = 20, c = 0;
	printf("sum()호출 전 a=%d b=%d c=%d\n", a, b, c);
	c = sum(a, b);
	printf("sum()호출 후 a=%d b=%d c=%d\n", a, b, c); 
	return 0;
}
int sum(int a, int b)
{
	a = a + 10;
	b = b + 20;
	printf("sum()함수 내 a=%d b=%d a+b=%d\n", a, b, a + b);
	return(a + b);
}

```



# 반복되는 부분은 함수로 구현

```c
#include <stdio.h>
void add(int x, int y);
void sub(int x, int y);
void mul(int x, int y);

int main(void)
{
	int num1 = 1, num2 = 2, num3 = 3;
	int num4 = 4, num5 = 5, num6 = 6;


	add(num1, num2);
	sub(num3, num4);
	mul(num5, num6);

	return 0;
}
void add(int x, int y)
{
	printf("두 수 %d과 %d를 더한 결과는 %d입니다.\n", x, y, x + y);
}

void sub(int x, int y)
{
	printf("두 수 %d과 %d를 뺀 결과는 %d입니다.\n", x, y, x - y);
}

void mul(int x, int y)
{
	printf("두 수 %d과 %d를 곱한 결과는 %d입니다.\n", x, y, x * y);
}

```

# auto변수와 static변수

```c++
#include <iostream>
void sub(void);
int main(void)
{
	sub();
	sub();
	sub();
	return 0;
}
void sub(void)
{
	using std::cout;
	using std::endl;
	int x = 10; //auto는 생략가능
	static int y = 10; //실행시 이 문장은 실행하지 않음, 컴파일시 이미 결정됨
	cout << "x=" << x << "y=" << y << endl;
	x++;
	y++;
}

```

# Man형 구조체를 선언하고 변수를 만들어서 멤버에 접근

```c++
#include <iostream>
struct Man { //구조체 선언
	char name[10]; //멤버
	int age; //멤버
	double weight; //멤버
};
int main(void)
{
	using std::cout;
	using std::endl;
	Man gildong, sunhee, comso[160];
	gildong.age = 20; //일반 구조체 변수의 멤버 참조
	sunhee.weight = 52.5;
	comso[0].age = 25;
	cout << gildong.age << endl << sunhee.weight << endl << comso[0].age;
	return 0;
}

```
